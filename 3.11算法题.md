## [49. 字母异位词分组 - 力扣（LeetCode）](https://leetcode.cn/problems/group-anagrams/?envType=study-plan-v2&envId=top-100-liked)

![image](https://github.com/user-attachments/assets/07e21e6c-9ca8-47eb-af44-68cab32dcc6c)

```cpp
class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        unordered_map<string,vector<string>> mp;//建立哈希表，键为拥有的字母，值为单词
        for(string& str : strs){//遍历字符串数组
            string key = str;
            sort(key.begin(),key.end());//让单词变成字母，保证不漏掉单词
            mp[key].emplace_back(str);//让拥有相同字母的单词放在值中
        }
        vector<vector<string>> res;//存放结果
        for(auto it = mp.begin(); it != mp.end(); ++it){
            res.emplace_back(it->second);//将值填入结果中
        }
        return res;
    }
};
```

## [128. 最长连续序列 - 力扣（LeetCode）](https://leetcode.cn/problems/longest-consecutive-sequence/submissions/705770397/?envType=study-plan-v2&envId=top-100-liked)

```cpp
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        unordered_set<int> num_set;//建立哈希表，去掉重复元素
        for(const int& num :nums){
            num_set.insert(num);
        }
        int res = 0;
        for(const int&num : num_set){
            if(!num_set.count(num - 1)){//当前数字没有与前面相连接的数字
                int cur = num;//让当前数字成为cur
                int curl = 1;//初始化当前序列长度
                while(num_set.count(cur + 1)){//寻找下一个连续数字
                    cur += 1;//更新当前数字
                    curl +=1;//更新序列长度
                }
                res = max(res,curl);//保证最大序列
            }
        }
        return res;
    }
};
```
