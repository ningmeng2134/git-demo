## [11. 盛最多水的容器 - 力扣（LeetCode）](https://leetcode.cn/problems/container-with-most-water/description/?envType=study-plan-v2&envId=top-100-liked)

![](https://raw.githubusercontent.com/ningmeng2134/git-demo/main/3.12/1-2026-03-12-17%3A19%3A54.png)

```cpp
class Solution {
public:
    int maxArea(vector<int>& height) {
        int i = 0;//初始化在最左边
        int r = height.size() - 1;//初始化到最右边
        int res = 0;//初始化
        while ( i < r){
            res = height[i] < height[r] ? max(res,(r - i)*height[i++]) : max(res,(r - i)*height[r--]);//更新最大面积
        }
        return res;
    }
};
```

## [15. 三数之和 - 力扣（LeetCode）](https://leetcode.cn/problems/3sum/description/?envType=study-plan-v2&envId=top-100-liked)

![](https://raw.githubusercontent.com/ningmeng2134/git-demo/main/3.12-2-2026-03-12-17%3A41%3A51.png)

```cpp
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        int n = nums.size();
        sort(nums.begin(),nums.end());//排序升序
        vector<vector<int>> res;//结果数组
        for(int a = 0; a < n; ++a){
            if(a > 0 && nums[a] == nums[a - 1]){
                continue;//防止产生重复三元组
            }
            int b = n - 1;//初始化为最右边
            int target = -nums[a];//b+c = -a
            for(int c = a + 1; c < n; ++c){
                if(c >a + 1 && nums[c] == nums[c - 1]){
                    continue;//防止重复
                }
                while(c < b && nums[c] + nums[b] > target){
                    --b;//当前值过大，让最右端的数向左移
                }
                if(c == b){
                    break;//没有可以匹配的数，提前结束循环
                }
                if(nums[c] + nums[b] == target){
                    res.push_back({nums[a],nums[b],nums[c]});//找到符合要求的元素
                }
            }
        }
        return res;
    }
};
```
