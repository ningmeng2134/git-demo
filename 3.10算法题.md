## [LCR 188. 买卖芯片的最佳时机 - 力扣（LeetCode）](https://leetcode.cn/problems/gu-piao-de-zui-da-li-run-lcof/)

```cpp
class Solution {
public:
    int bestTiming(vector<int>& prices) {
        int cost = INT_MAX, profit = 0;//初始化一个很大的cost保证最低花费、一个很小的profit保证最大利润
        for(int price : prices){//遍历prices数组
            cost = min(cost,price);//更新花费与利润
            profit = max(profit,price - cost);//转移方程
        }
        return profit;
    }
};
```

## [233. 数字 1 的个数 - 力扣（LeetCode）](https://leetcode.cn/problems/number-of-digit-one/description/?envType=problem-list-v2&envId=8LSpuXqD)

```cpp
class Solution {
public:
    int countDigitOne(int n) {
        long digit = 1;//位数
        int high = n / 10；//高位
        int cur = n % 10;//当前位
        int low = 0;//低位
        int res = 0;//结果
        while(high != 0 || cur != 0){//保证所有数位都可以被循环到
            //情况1当前位数字为0 1的出现次数由high决定 high*digit
            if(cur == 0) res += high * digit;
            //情况2当前位数字为1 1出现次数由high和low决定 high*dight+low+1
            else if(cur == 1) res += high * digit + low + 1;
            //其他情况由high决定 （high+1）*dight
            else res += (high + 1) * digit;
            //更新低位当前位高位位数
            low += cur * digit;
            cur = high % 10;
            high /= 10;
            digit *= 10;
        }
        return res;
    }
};
```
